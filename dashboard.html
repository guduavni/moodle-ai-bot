function extractQuestion(text) { if (!text) return ""; let t = String(text).trim(); // ניקוי רווחים ושורות מיותרות t = t.replace(/\s+/g, " "); // אם הטקסט כולל פתיח הדרכתי, נחתוך ממנו רק אחרי "שאלה:" const questionMarkers = [ "שאלה:", "השאלה:", "Question:", "question:" ]; for (const marker of questionMarkers) { const index = t.indexOf(marker); if (index >= 0) { t = t.substring(index + marker.length).trim(); break; } } // חיתוך כל מה שמתחיל באפשרויות תשובה const answerMarkers = [ "אפשרויות תשובה:", "אפשרויות תשובה", "אפשרויות:", "תשובות:", "אפשרויות התשובה:", "אפשרויות התשובה", "Answer choices:", "Choices:", "Options:" ]; for (const marker of answerMarkers) { const index = t.indexOf(marker); if (index > 0) { t = t.substring(0, index).trim(); break; } } // חיתוך לפי תחילת אפשרות ממוספרת, רק אם היא נראית כמו תשובה t = t.replace(/\s+(?:1\.|1\)|א\.|א\)|A\.|A\))\s+.+$/s, "").trim(); // הסרת משפטים שמגיעים לפני השאלה בפורמטים נפוצים t = t.replace(/^.*?שאלה[:：]\s*/s, "").trim(); // ניקוי סימני סיום מיותרים t = t.replace(/\s+$/g, "").trim(); return t; }
